[docs]class NullFiller(BaseEstimator, NoFitMixin, TransformerMixin):
    def __init__(self, isnull=None, replacement=0.0):
        super().__init__()
        if isnull is None:
            self.isnull = np.isnan
        else:
            self.isnull = isnull
        self.replacement = replacement


[docs]    def transform(self, X, **transform_kwargs):
        X = X.copy()
        mask = self.isnull(X)
        X[mask] = self.replacement
        return X
