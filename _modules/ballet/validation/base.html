def check_from_class(check_class, obj, *checker_args, **checker_kwargs):
    for Checker in check_class.__subclasses__():
        check = Checker(*checker_args, **checker_kwargs).do_check
        name = Checker.__name__
        success = check(obj)
        if not success:
            yield name
